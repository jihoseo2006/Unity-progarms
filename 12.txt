#define _CRT_SECURE_NO_WARNINGS //14ë²ˆ
#include <stdio.h>

int main()
{
    int num;
    scanf("%d", &num);
    for (int i = 1; i <= num; i++) 
    {
        for (int j = 0; j < num - i; j++) {
            printf(" ");
        }
        for (int j = 1; j <= i; j++) {
            printf("*");
        }
            printf("\n");
    }
    for (int i = 2; i <= num; i++) {
        for (int j = 0; j < i - 1; j++) {
            printf(" ");
        }
        for (int j = num; i <= j; --j) {
            printf("*");
        }
        printf("\n");
    }
    for (int i = 1; i <= num; i++) {
        for (int j = 1; j <= i; j++) {
            printf("*");
        }
        printf("\n");
    }
    for (i = num-1; i >= 1; i--)
    {
        for (j = 1; j <= i; j++) {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}
